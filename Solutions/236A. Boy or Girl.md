## ğŸ§± Variant 1 â€” Manual Filtering (vector + find)

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    string s;
    cin >> s;
    vector<char> v;

    for (char c : s) {
        if (find(v.begin(), v.end(), c) == v.end()) {
            v.push_back(c);
        }
    }

    cout << (v.size() % 2 == 0 ? "CHAT WITH HER!" : "IGNORE HIM!") << endl;
    return 0;
}
